#include <bits/stdc++.h>
using namespace std;

vector <int> solve(vector <int>& a, int k) {
    deque <int> d;
    vector <int> ans;
    for (int i = 0; i < a.size(); i++) {
        if (!d.empty() && d.front() == i - k)
            d.pop_front();
        while (!d.empty() && a[d.back()] >= a[i])
            d.pop_back();
        d.push_back(i);
        if (i >= k - 1)
            ans.push_back(a[d.front()]);
    }
    return ans;
}

int main()
{
    vector <int> a = {1, 3, -1, -3, 5, 3, 6, 7};
    int k = 3;
    vector <int> ans = solve(a, k);
    for (int i : ans)
        cout << i << " ";

    return 0;
}
