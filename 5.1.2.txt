#include <bits/stdc++.h>
using namespace std;

ifstream f("strabunica.in");
ofstream g("strabunica.out");

int n, st[200002], dr[200002];
long long ans, h[200002];
stack <int> s, s1;
int main()
{
    f >> n;
    for (int i = 1; i <= n; i++)
        f >> h[i];
    s.push(n + 1);
    for (int i = n; i > 0; i--) {
        while (!s.empty() && h[s.top()] >= h[i])
            s.pop();
        if (!s.empty())
            dr[i] = s.top();
        s.push(i);
    }
    s1.push(0);
    for (int i = 1; i <= n; i++) {
        while(!s1.empty() && h[s1.top()] >= h[i])
            s1.pop();
        if(!s1.empty())
            st[i] = s1.top();
        s1.push(i);
    }
    for (int i = 1; i <= n; i++)
        ans = max(ans, h[i] * (dr[i] - st[i] - 1));
    g << ans;

    return 0;
}
