#include <bits/stdc++.h>
using namespace std;

int n, a[100002], target, j, s;
int main()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    cin >> target;
    j = n;
    for (int i = 1; i <= n; i++) {
        while (a[i] + a[j] > target && j > i)
            j--;
        if (a[i] + a[j] == target && i < j) {
            cout << '[' << i << ',' << " " << j << ']';
            return 0;
        }
    }
}
