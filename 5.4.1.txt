#include <bits/stdc++.h>
using namespace std;

int n;
long long a[100002], st = 1, dr = 1, ans, k;

bool verif(long long x) {
    long long cnt = 0;
    for (int i = 1; i <= n; i++)
        cnt = cnt + a[i] / x;
    return cnt >= k;
}

int main()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> a[i], dr = max(dr, a[i]);
    cin >> k;
    while (st <= dr) {
        long long mij = (st + dr) / 2;
        if (verif(mij))
            ans = mij, st = mij + 1;
        else
            dr = mij - 1;
    }
    cout << ans;
}
