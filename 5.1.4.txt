bool knows(int a, int b);

int Find(int n) {
    int candidat = 0;
    for (int i = 1; i < n; i++)
        if (knows(candidat, i))
            candidat = i;
    for (int i = 0; i < n; i++) {
        if (i == candidat)
            continue;
        if (knows(candidat, i) || !knows(i, candidat))
            return -1;
    }
    return candidat;
}
