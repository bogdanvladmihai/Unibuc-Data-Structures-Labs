#include <bits/stdc++.h>
using namespace std;

int n, m, x, i, j;
long long sp[100002], ans = INT_MIN;
int main()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> x, sp[i] = sp[i - 1] + x;
    cin >> m;
    while (m--) {
        cin >> i >> j;
        if (i > j)
            swap(i, j);
        ans = max(ans, sp[j] - sp[i - 1]);
    }
    cout << ans;
}
