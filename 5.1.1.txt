#include <bits/stdc++.h>
using namespace std;

int n, x;
stack <pair<int, int>> s;
int main()
{
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> x;
        while (!s.empty()) {
            if (s.top().first >= x)
                s.pop();
            else {
                cout << s.top().second << " ";
                s.push(make_pair(x, i));
                break;
            }
        }
        if (s.empty()) {
            cout << 0 << " ";
            s.push(make_pair(x, i));
        }
    }

    return 0;
}
